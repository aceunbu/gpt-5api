def analyze_image_with_gpt4v(image_url_or_base64, prompt):
    messages = [
        {"role": "user", "content": [
            {"type": "text", "text": prompt}
        ]}
    ]
    
    if image_url_or_base64.startswith("http"):
        messages[0]["content"].append({
            "type": "image_url",
            "image_url": {"url": image_url_or_base64}
        })
    else: # Assume base64
        messages[0]["content"].append({
            "type": "image_url",
            "image_url": {"url": f"data:image/jpeg;base64,{image_url_or_base64}"}
        })

    data = {
        "model": "gpt-4-vision-preview", # 或 gpt-4o
        "messages": messages,
        "max_tokens": 500
    }
    
    response = requests.post(API_URL_VISION, headers=headers, json=data, timeout=30)
    response.raise_for_status()
    return response.json()
